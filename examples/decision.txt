# decision.txt
# ProcessLang Example: Making Difficult Decisions

# Use case: Facing a choice between multiple options
# ProcessLang approach: Choice as collapse, not comparison

FLOW engagement=0.8 resistance=0.2
# High engagement with some acknowledged resistance
# Decisions feel heavy because they collapse possibilities

ENCODE type="category" loss=0.4
# Categorize the decision space
# Accept moderate information loss
# Not all factors can be quantified

CHOOSE space=["option_a", "option_b", "option_c"] method="conscious"
# Explicit choice mechanism
# Conscious method: Deliberate selection
# The act of choosing collapses the possibility space

DISSOLVE rigidity=0.6 target="unchosen paths"
# Dissolve attachment to unchosen possibilities
# Moderate rigidity: some mourning is natural
# But must dissolve to move forward

CYCLE mode="stable" intensity=0.7
# Stable cycle with moderate intensity
# Decision made, now reinforce the path chosen
# Cycle creates momentum

QUERY: What opens up on this path that wasn't visible before choice?
# Query focuses on emergence after collapse
# Not "did I choose right?" but "what emerges from this choice?"
